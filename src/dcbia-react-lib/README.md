## Install

```bash
npm install --save dcbia-react-lib
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'dcbia-react-lib'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
